Ejercicio 2.7.7 – Matriz inversa


% Ejercicio 2.7.7
A = [3 6; 2 8];
B = [-2  3  6;
      4  8 -5;
      9 -9  7];
C = [-3  3  6  1;
      4  2  8 -5;
      6 -6  7  2;
      7  2 -2  9];

Inversa_A = inv(A);
Inversa_B = inv(B);
Inversa_C = inv(C);

disp('Inversa de A ='); disp(Inversa_A);
disp('Inversa de B ='); disp(Inversa_B);
disp('Inversa de C ='); disp(Inversa_C);
