Ejercicio 2.7.6 – Matriz adjunta (adjugate)


% Ejercicio 2.7.6
A = [3 6; 2 8];
B = [-2  3  6;
      4  8 -5;
      9 -9  7];

% Adjunta (adjugate): adj(M) = det(M) * inv(M)
Adjunta_A = det(A) * inv(A);
Adjunta_B = det(B) * inv(B);

disp('Adjunta de A ='); disp(Adjunta_A);
disp('Adjunta de B ='); disp(Adjunta_B);
