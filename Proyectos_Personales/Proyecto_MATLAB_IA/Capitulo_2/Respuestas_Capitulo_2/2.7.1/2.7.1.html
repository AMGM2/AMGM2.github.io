Ejercicio 2.7.1 – Producto punto con transpuestas


% Ejercicio 2.7.1
a  = [  2; -3;  5; -6];
b  = [ -7; -4; -12;  2];

aT = a.';       % transpuesta de a (fila)
bT = b.';       % transpuesta de b (fila)

Ra = aT * b;    % a^T b
Rb = bT * a;    % b^T a  (debe dar lo mismo)

disp('a^T ='), disp(aT)
disp('b^T ='), disp(bT)
disp('a^T b ='), disp(Ra)
disp('b^T a ='), disp(Rb)
