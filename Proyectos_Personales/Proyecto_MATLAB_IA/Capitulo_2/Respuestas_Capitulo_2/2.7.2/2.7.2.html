Ejercicio 2.7.2 – Transpuesta de matrices (ejemplos + entrada del usuario)


% Ejercicio 2.7.2
% Ejemplos
A = [ -2  6 -1;  -5 -2  4 ];    % 2x3
B = [  2  4 -3;   4  7 -7;  -5  1  3 ];  % 3x3

AT = A.';    % transpuesta
BT = B.';

disp('AT ='), disp(AT)
disp('BT ='), disp(BT)

% Programa breve: transpuesta de una matriz introducida por el usuario
disp('% Programa para obtener la transpuesta de cualquier matriz');
disp('Ingrese la matriz A entre corchetes. Ejemplo: [1 2; 3 4]');
Ausr = input('Matriz A: ');
AusrT = Ausr.';
disp('La transpuesta de A es:');
disp(AusrT);
