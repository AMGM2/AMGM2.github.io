Ejercicio 2.7.9 – Mínimos cuadrados con 2 variables y predicción


% Ejercicio 2.7.9
% Conjunto de datos (X con término de sesgo y dos variables; Y como salida)
% (x1,y1)=([2,3],4), (x2,y2)=([3,3],3), (x3,y3)=([2,4],3), (x4,y4)=([3,4],5)

format short
X = [ 1 2 3;
      1 3 3;
      1 2 4;
      1 3 4 ];   % [1, x1, x2]
Y = [4; 3; 3; 5];

% Parámetros por mínimos cuadrados: beta = (X'X)^-1 X'Y
beta = (X' * X) \ (X' * Y);
disp('beta ='); disp(beta);

% Estimar Y para x = [5; 6]:
x_nueva = [1; 5; 6];
y_estimada = beta' * x_nueva;
disp('y estimada para x = [5;6] ='); disp(y_estimada);
